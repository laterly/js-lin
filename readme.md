	import { publish } from 'js-lin';
	
	function event(data) {
            console.log(data);
        }
    //订阅事件a
    publish.on('a', (data) => {
        event(data)
    });
    //触发事件a
    publish.emit('a', '我是第1次调用的参数');
	
	//在vue中Index.vue和Main.vue通信
	Index.vue:
	export default {
	created() {
		//订阅事件a
    	publish.on('a', this.event);
 	},
	methods: {
		event(data){
			console.log('事件'+data);
		}
  	},
	beforeDestroy() {
		publish.remove('a', this.event);
 	},
	};
	Main.vue:
	export default {
	created() {
		this.emitEvent();
 	},
	methods: {
		emitEvent(){
			publish.emit('a', '参数');
		}
  	}
	};
</pre>
